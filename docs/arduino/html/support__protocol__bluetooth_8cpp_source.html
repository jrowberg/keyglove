63 uint8_t process_protocol_command_bluetooth(uint8_t *rxPacket) {
64  // check for valid command IDs
65  uint8_t protocol_error = 0;
66  switch (rxPacket[3]) {
67  case KG_PACKET_ID_CMD_BLUETOOTH_GET_MODE: // 0x01
68  // bluetooth_get_mode()(uint16_t result, uint8_t mode)
69  // parameters = 0 bytes
70  if (rxPacket[1] != 0) {
71  // incorrect parameter length
72  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
73  } else {
74  // run command
75  uint8_t mode;
76  uint16_t result = kg_cmd_bluetooth_get_mode(&mode);
77 
78  // build response
79  uint8_t payload[3] = { result & 0xFF, (result >> 8) & 0xFF, mode };
80 
81  // send response
82  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 3, rxPacket[2], rxPacket[3], payload);
83  }
84  break;
85 
86  case KG_PACKET_ID_CMD_BLUETOOTH_SET_MODE: // 0x02
87  // bluetooth_set_mode(uint8_t mode)(uint16_t result)
88  // parameters = 1 byte
89  if (rxPacket[1] != 1) {
90  // incorrect parameter length
91  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
92  } else {
93  // run command
94  uint16_t result = kg_cmd_bluetooth_set_mode(rxPacket[4]);
95 
96  // build response
97  uint8_t payload[2] = { result & 0xFF, (result >> 8) & 0xFF };
98 
99  // send response
100  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 2, rxPacket[2], rxPacket[3], payload);
101  }
102  break;
